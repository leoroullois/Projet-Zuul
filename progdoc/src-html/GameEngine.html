123    /**
124     * Affiche le message de bienvenue
125     */
126    private void printWelcome() {
127        this.aGui.println("Welcome to the Trade Infinity Game!");
128        this.aGui.println("World of Blockchains is a new, incredibly boring trading and investing game.");
129        String vAllRooms = "All rooms :";
130        Set<String> allKeys = this.aAllRooms.keySet();
131        for (String vKey : allKeys) {
132            vAllRooms += " " + vKey;
133        }
134        this.aGui.println(vAllRooms);
135        this.aGui.println("Type 'help' if you need help.");
136        this.printLocationInfo();
137        if (this.aPlayer.getCurrentRoom().getImageName() != null) {
138            this.aGui.showImage(this.aPlayer.getCurrentRoom().getImageName());
139        }
140    }

142    /**
143     * Check si la commande est valide, si oui return true sinon return false
144     * 
145     * @param pCommand la commande a check
146     * @return true ou false en fonction de si la commande est valide
147     */
148    public void interpretCommand(final String pCommandLine) {
149        this.aGui.println("> " + pCommandLine);
150        Command vCommand = this.aParser.getCommand(pCommandLine);
151
152        if (vCommand.isUnknown()) {
153            this.aGui.println("I don't know what you mean...");
154        } else {
155            if (vCommand.getCommandWord().equals("go")) {
156                this.goRoom(vCommand);
157            } else if (vCommand.getCommandWord().equals("help")) {
158                this.printHelp();
159            } else if (vCommand.getCommandWord().equals("look")) {
160                this.look(vCommand);
161            } else if (vCommand.getCommandWord().equals("buy")) {
162                this.buy(vCommand);
163            } else if (vCommand.getCommandWord().equals("back")) {
164                this.back(vCommand);
165            } else if (vCommand.getCommandWord().equals("test")) {
166                this.test(vCommand);
167            } else if (vCommand.getCommandWord().equals("take")) {
168                this.take(vCommand);
169            } else if (vCommand.getCommandWord().equals("drop")) {
170                this.drop(vCommand);
171            } else if(vCommand.getCommandWord().equals("items")) {
172                this.items(vCommand);
173            } else if (vCommand.getCommandWord().equals("eat")) {
174                this.eat(vCommand);
175            }
176            if (vCommand.getCommandWord().equals("quit")) {
177                this.quit(vCommand);
178            }
179        }
180    }

197    /**
198     * Défini l'action de la commande look :
199     * look : donne des informations sur la salle actuelle, les items présents à l'intérieur et les sorties.
200     * look wallet : affiche l'argent du joueur.
201     * @param pCommand commande look
202     */
203    private void look(final Command pCommand) {
204        if (pCommand.hasSecondWord()) {
205            if (pCommand.getSecondWord().equals("wallet")) {
206                this.aGui.println(this.aPlayer.getBalanceString());
207            } else {
208                Boolean vTest = false;
209                Set<String> allKeys = this.aPlayer.getCurrentRoom().getItems().keySet();
210                for (String vKey : allKeys) {
211                    if (vKey.equals(pCommand.getSecondWord())) {
212                        vTest = true;
213                        this.aGui
214                                .println("This is "
215                                        + this.aPlayer.getCurrentRoom().getItems().get(vKey).getDescription());
216                    }
217                }
218                if (!vTest) {
219                    this.aGui.println("I don't know what you are looking for.");
220                }
221            }
222        } else {
223            this.aGui.println(this.aPlayer.getCurrentRoom().getLongDescription());
224        }
225    }

253    /**
254     * Permet de se déplacer
255     * 
256     * @param pCommand commande pour se déplacer, par exemple "go south"
257     */
258    private void goRoom(final Command pCommand) {
259        if (!pCommand.hasSecondWord()) {
260            this.aGui.println("Go where ?");
261            return;
262        }
263
264        String vDirection = pCommand.getSecondWord();
265        Room vNextRoom = aPlayer.getCurrentRoom().getExit(vDirection);
266        if (vNextRoom == null) {
267            this.aGui.println("There is no door !");
268            return;
269        } else {
270            this.aPlayer.changeRoom(vNextRoom);
271            this.printLocationInfo();
272            if (this.aPlayer.getCurrentRoom().getImageName() != null) {
273                this.aGui.showImage(this.aPlayer.getCurrentRoom().getImageName());
274            }
275        }
276    }

278    /**
279     * Permet de retourner dans les salles précédentes
280     * 
281     * @param pCommand commande tapée au clavier
282     */
283    private void back(final Command pCommand) {
284        if (pCommand.hasSecondWord()) {
285            this.aGui.println("I don't understand what you are saying.");
286        } else {
287            if (!this.aPlayer.getPrevRooms().empty()) {
288                this.aPlayer.goBack();
289                this.printLocationInfo();
290            } else {
291                this.aGui.println("There are no previous room.");
292            }
293            if (this.aPlayer.getCurrentRoom().getImageName() != null) {
294                this.aGui.showImage(this.aPlayer.getCurrentRoom().getImageName());
295            }
296        }
297    }
298    /**
299     * Test le jeu à partir d'un fichier texte contenant les commandes voulues.
300     * @param pCommand commande test fichier pour tester le jeu avec les commandes dans fichier.txt
301     */
302    private void test(final Command pCommand) {
303        if (!pCommand.hasSecondWord()) {
304            this.aGui.println("What file do you want for the test ?");
305        } else {
306            Scanner myFile = null;
307            try {
308                myFile = new Scanner(new BufferedReader(new FileReader("tests/" + pCommand.getSecondWord() + ".txt")));
309                while (myFile.hasNext()) {
310                    String aLigne = myFile.nextLine();
311                    this.interpretCommand(aLigne);
312                }
313            } catch (final FileNotFoundException e) {
314                this.aGui.println("Error : " + e.getMessage());
315            } finally {
316                if (myFile != null) {
317                    myFile.close();
318                }
319            }
320        }
321    }
