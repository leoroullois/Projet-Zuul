118    /**
119     * Affiche le message de bienvenue
120     */
121    private void printWelcome() {
122        this.aGui.println("Welcome to the Trade Infinity Game!");
123        this.aGui.println("World of Blockchains is a new, incredibly boring trading and investing game.");
124        String vAllRooms = "All rooms :";
125        Set<String> allKeys = this.aAllRooms.keySet();
126        for (String vKey : allKeys) {
127            vAllRooms += " " + vKey;
128        }
129        this.aGui.println(vAllRooms);
130        this.aGui.println("Type 'help' if you need help.");
131        this.printLocationInfo();
132        if (this.aCurrentRoom.getImageName() != null) {
133            this.aGui.showImage(this.aCurrentRoom.getImageName());
134        }
135    }
136
137    /**
138     * Check si la commande est valide, si oui return true sinon return false
139     * 
140     * @param pCommand la commande a check
141     * @return true ou false en fonction de si la commande est valide
142     */
143    public boolean interpretCommand(final String pCommandLine) {
144        this.aGui.println("> " + pCommandLine);
145        Command vCommand = this.aParser.getCommand(pCommandLine);
146
147        if (vCommand.isUnknown()) {
148            this.aGui.println("I don't know what you mean...");
149            return false;
150        } else {
151            if (vCommand.getCommandWord().equals("go")) {
152                this.goRoom(vCommand);
153            } else if (vCommand.getCommandWord().equals("help")) {
154                this.printHelp();
155            } else if (vCommand.getCommandWord().equals("look")) {
156                this.look(vCommand);
157            } else if (vCommand.getCommandWord().equals("buy")) {
158                this.buy(vCommand);
159            } else if (vCommand.getCommandWord().equals("back")) {
160                this.back(vCommand);
161            }
162
163            if (vCommand.getCommandWord().equals("quit")) {
164                return quit(vCommand);
165            } else {
166                return false;
167            }
168        }
169    }

171    /**
172     * Permet de quitter le jeu
173     * 
174     * @param pQuit commande quit
175     * @return true si on quitte le jeu sinon false
176     */
177    private boolean quit(final Command pQuit) {
178        if (pQuit.hasSecondWord()) {
179            this.aGui.println("Quit what ??");
180            return false;
181        } else {
182            this.endGame();
183            return true;
184        }
185    }

187    /**
188     * Défini l'action de la commande look
189     * 
190     * @param pCommand commande look
191     */
192    private void look(final Command pCommand) {
193        if (pCommand.hasSecondWord()) {
194            Boolean vTest = false;
195            Set<String> allKeys = this.aCurrentRoom.getItems().keySet();
196            for (String vKey : allKeys) {
197                if (vKey.equals(pCommand.getSecondWord())) {
198                    vTest = true;
199                    this.aGui.println("This is  " + this.aCurrentRoom.getItems().get(vKey).getDescription());
200                }
201            }
202            if (!vTest) {
203                this.aGui.println("I don't know what you are looking for.");
204            }
205        } else {
206            this.aGui.println(this.aCurrentRoom.getLongDescription());
207        }
208    }

231    /**
232     * Permet de se déplacer
233     * 
234     * @param pCommand commande pour se déplacer, par exemple "go south"
235     */
236    private void goRoom(final Command pCommand) {
237        if (!pCommand.hasSecondWord()) {
238            this.aGui.println("Go where ?");
239            return;
240        }
241
242        String vDirection = pCommand.getSecondWord();
243        Room vNextRoom = aCurrentRoom.getExit(vDirection);
244        if (vNextRoom == null) {
245            this.aGui.println("There is no door !");
246            return;
247        } else {
248            this.aPrevRooms.push(this.aCurrentRoom);
249            this.aCurrentRoom = vNextRoom;
250            this.printLocationInfo();
251        }
252        if (this.aCurrentRoom.getImageName() != null) {
253            this.aGui.showImage(this.aCurrentRoom.getImageName());
254        }
255    }
256
257    /**
258     * Permet de retourner dans les salles précédentes
259     * @param pCommand  commande tapée au clavier
260     */
261    private void back(final Command pCommand) {
262        if (pCommand.hasSecondWord()) {
263            this.aGui.println("I don't understand what you are saying.");
264        } else {
265            if (!this.aPrevRooms.empty()) {
266                this.aCurrentRoom=this.aPrevRooms.pop();
267                this.printLocationInfo();
268            } else {
269                this.aGui.println("There are no previous room.");
270            }
271            if (this.aCurrentRoom.getImageName() != null) {
272                this.aGui.showImage(this.aCurrentRoom.getImageName());
273            }
274        }
275    }
276    private void endGame() {
277        this.aGui.println("Thank you for playing.  Good bye.");
278        this.aGui.enable(false);
279    }
